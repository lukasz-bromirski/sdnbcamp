!
! BGP IPv4 configuration for Nexus 9000 LEAF
!

router bgp 6500X
 address-family ipv4 unicast
 address-family ipv6 unicast
 template peer BGP-RR
   remote-as 6500X
   update-source loopback0
   address-family ipv4 unicast
   address-family ipv6 unicast
 neighbor 10.1X.192.1
  inherit peer BGP-RR
 neighbor 10.1X.192.2
  inherit peer BGP-RR
